                        function aclean(arr) {
                            let map = new Map();
                        
                            for (let el of arr) {
                                let sorted = el.toLowerCase().split("").sort().join("");
                                map.set(sorted, el)
                            }
                        
                            return Array.from(map.values());
                        }
